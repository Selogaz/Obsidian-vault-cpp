
```c++
#include <fstream>
#include <iostream>

using namespace std;

class Point {
public:
	int x;
	int y;
	int z;

	void Print() {
		cout << "\tx:"<<x <<"\ty:" <<y <<"\tz:"<< z << endl;
	}

	Point() {
		x=y=z=0;
	}
	Point(x,y,z) {
		this->x = x;
		this->y = y;
		this->z = z;
	}
	~Point() {
	}
};

int main() {
	ofstream fout;
	string path = "myFile.txt";
	fout.open(path, ofstream::app);//дописать в конец файла. Без второго аргумента будет стирать все, что было до
	if (!fout.is_open()) {
		cout << "Ошибка открытия файла!" << endl;
	}
	else {
		Point point(155,255,0);
		fout.write((char*)&point, sizeof(Point));
		cout << "Запись проведена успешно" << endl;
	}
	fout.close();
	
	ifstream fin;
	fin.open(path);
	if (!fin.is_open()) {
		cout << "Ошибка открытия файла" << endl;
	}
	else {
		cout << "Файл открыт" << endl;
		Point pnt;
		while (fin.read((char*)&pnt, sizeof(Point))) {
			pnt.Print();
		}
	}
	fin.close();
}
```
