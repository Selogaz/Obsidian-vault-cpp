## Примеры задач

%% Задача 1
$$
\begin{aligned}
& \frac{1}{3x-4} = 5; \quad x \neq \frac{4}{3} \\
& 1 = 5(3x-4) \\
& 1 = 15x - 20 \\
& 21 = 15x \\
& x = \frac{21}{15} = \frac{7}{5} = 1,4
\end{aligned}
$$

%% Задача 2
$$
\begin{aligned}
& (x+4)^3 = -125 \\
& \text{Нечетные степени - убираем} \\
& (x+4)^3 = (-5)^3 \\
& x + 4 = -5 \\
& x = -9
\end{aligned}
$$

%% Задача 3
$$
\begin{aligned}
& \sqrt[3]{x+6} = 4 \\
& x + 6 = 64 \\
& x = 58
\end{aligned}
$$

%% Задача 4
$$
\begin{aligned}
& \sqrt{9x-47} = 4 \\
& 9x - 47 = 16 \\
& 9x = 63 \\
& x = 7
\end{aligned}
$$

%% Задача 5
$$
\begin{aligned}
& \left(\frac{1}{7}\right)^{x+4} = 49 \\
& (7^{-1})^{x+4} = 7^2 \implies 7^{-1(x+4)} = 7^2 \\
& -x - 4 = 2 \\
& -x = 6 \\
& x = -6
\end{aligned}
$$

%% Задача 6
$$
\begin{aligned}
& 2^{-4-x} = 16 \\
& 2^{-4-x} = 2^4 \\
& -4 - x = 4 \\
& x = -8
\end{aligned}
$$

%% Задача 7
$$
\begin{aligned}
& \left(\frac{1}{6}\right)^{x-2} = 6^x \\
& (6^{-1})^{x-2} = 6^x \\
& -x + 2 = x \\
& 2x = 2 \\
& x = 1
\end{aligned}
$$
%% Задача 8
$$
\begin{aligned}
& 5^{2-x} = 125 \\
& 5^{2-x} = 5^3 \\
& 2 - x = 3 \\
& x = -1
\end{aligned}
$$

%% Задача 9
$$
\begin{aligned}
& 3^{x-8} = \frac{1}{81} \\
& 3^{x-8} = \frac{1}{3^4} = 3^{-4} \\
& x - 8 = -4 \\
& x = 4
\end{aligned}
$$

%% Задача 10
$$
\begin{aligned}
& \left(\frac{1}{6}\right)^{x-3} = \frac{1}{36} \\
& \left(\frac{1}{6}\right)^{x-3} = \left(\frac{1}{6}\right)^2 \\
& x - 3 = 2 \\
& x = 5
\end{aligned}
$$

%% Задача 11
$$
\begin{aligned}
& 3^{x+6} = 9^{2x} \\
& 3^{x+6} = (3^2)^{2x} = 3^{4x} \\
& x + 6 = 4x \\
& 3x = 6 \\
& x = 2
\end{aligned}
$$
%% Задача 12
$$
\begin{aligned}
& \sqrt{63-9x} = 3 \\
& 63-9x = 9 \\
& 54 = 9x \\
& x = 6
\end{aligned}
$$

%% Задача 13
$$
\begin{aligned}
& \log_5(8-x) = \log_5 2 \quad (\text{монотонность ф-ции}) \\
& 8 - x = 2 \\
& x = 6
\end{aligned}
$$

%% Задача 14
$$
\begin{aligned}
& \log_5(20-x) = 2 \\
& \log_5(20-x) = \log_5 25 \\
& 20 - x = 25 \\
& x = -5 \\
& \text{или потенцирование: } 5^2 = 20 - x \implies x = -5
\end{aligned}
$$

%% Задача 15
$$
\begin{aligned}
& \sqrt{15-2x} = x \\
& 15 - 2x = x^2 \\
& x^2 + 2x - 15 = 0 \\
& \text{По теореме Виета:} \\
& x_1 + x_2 = -2 \\
& x_1 \cdot x_2 = -15 \\
& x_1 = -5, \quad x_2 = 3 \\
& D = b^2 - 4ac = 4 - 4 \cdot 1 \cdot (-15) = 64 \\
& x_{1,2} = \frac{-2 \pm 8}{2} \\
& x_1 = -5 \quad (\text{не подходит, т.к. } x \ge 0) \\
& x_2 = 3 \quad (\text{ответ})
\end{aligned}
$$
%% Справочная заметка
$$ \sqrt{f} = g \implies f = g^2, \quad g \ge 0 $$

%% Задача 16
$$
\begin{aligned}
& \sqrt{72+x} = -x \\
& \begin{cases} 72+x = (-x)^2 \\ -x \ge 0 \end{cases} \implies \begin{cases} x^2 - x - 72 = 0 \\ x \le 0 \end{cases} \\
& x_1 = 9 \\
& x_2 = -8 \quad (\text{подходит, т.к. } -8 \le 0)
\end{aligned}
$$

%% Задача 17
$$
\begin{aligned}
& 2^{3+x} = 0,4 \cdot 5^{3+x} \\
& 2^{3+x} = \frac{2}{5} \cdot 5^{3+x} \\
& \text{Показательная ф-ция всегда } > 0 \\
& \left(\frac{2}{5}\right)^{3+x} = \left(\frac{2}{5}\right)^1 \\
& 3 + x = 1 \\
& x = -2
\end{aligned}
$$
