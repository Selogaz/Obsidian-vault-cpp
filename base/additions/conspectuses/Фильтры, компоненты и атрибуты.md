# Условия

## V-if
```html
<div id="app">
	<h1 v-if="5 > 5">{{ title }}</h1>
</div>
```
```js
const { createApp } = Vue

createApp({
    data() {
        return {
            title: "Hello World"
        }
    }
}).mount("#app")
```

## V-else
```html
<h1 v-if="title != 'Hello World'">{{ title }}</h1>
<p v-else><b>{{ title }}</b></p>
```

## V-else-if
```html
<h1 v-if="title != 'Hello World'">{{ title }}</h1>
        <p v-else-if="title == 'Hello World'"><em>{{ title }}</em></p>
        <p v-else><b>{{ title }}</b></p>
```

## Кнопка скрыть/показать

### V-if - удаляет элемент с разметки
```html
<button @click="isVisible = !isVisible">Скрыть или показать</button>
<p v-if="isVisible">Некий текст</p>
```

```js
const { createApp } = Vue

createApp({
    data() {
        return {
            title: "Hello World",
            isVisible: true
        }
    }
}).mount("#app")
```

### V-show - просто скрывает

добавляет свойство `display:none`
```html
<button @click="isVisible = !isVisible">Скрыть или показать</button>
<p v-show="isVisible">Некий текст</p>
```

# Циклы

## Изначальный массив
```js
const { createApp } = Vue

createApp({
    data() {
        return {
            title: "Hello World",
            isVisible: true,
            books: [
                {name: '1984', author: 'Джордж Оруэлл', genre: 'Роман'},
                {name: 'Собака Баскервилей', author: 'Артур Конан Дойл', genre: 'Детектив'},
                {name: 'Гарри Поттер', author: 'Джоан Роулинг', genre: 'Роман'}
            ]
        }
    }
}).mount("#app")
```

### Доступ к элементам массива
```html
<div id="app">
	{{ books[1].author }}
</div>
```

## V-for
```html
<div id="app">
	<ul>
		<li v-for="el in books">{{ el.name }} - <b>{{ el.author }}</b> / <em>{{ el.genre }}</em></li>
	</ul>
</div>
```

### Получение индекса
```html
<div id="app">
	<ul>
		<li v-for="(el,i) in books">{{ i + 1}} - {{ el.name }} - <b>{{ el.author }}</b> / <em>{{ el.genre }}</em></li>
	</ul>
</div>
```

### Computed[^2]
```js
const { createApp } = Vue

createApp({
    data() {
        return {
            title: "Hello World",
            isVisible: true,
            books: [
                {name: '1984', author: 'Джордж Оруэлл', genre: 'Роман'},
                {name: 'Собака Баскервилей', author: 'Артур Конан Дойл', genre: 'Детектив'},
                {name: 'Гарри Поттер', author: 'Джоан Роулинг', genre: 'Роман'}
            ]
        }
    },
    computed: {
        booksLowercase() {
            return this.books.map(book => ({
                ...book,
                name: book.name.toLowerCase()
            }));
        }
    }
}).mount("#app")
```
```html
<ul>
    <li v-for="(el, i) in booksLowercase">{{ i + 1}} - {{ el.name }} - <b>{{ el.author }}</b> / <em>{{ el.genre }}</em></li>
</ul>
```

# Компоненты Vue js

Компонент - независимый блок

## Пример создания компонента

В уроке снова vue2, так что пришлось переделать под vue3[^3]
Вырезали из прошлого html тег `ul` и вставили в `template`, предварительно убрав все переносы строк.
```js
const { createApp } = Vue;

const MainComponent = {
    data() {
        return {
            books: [
                {name: '1984', author: 'Джордж Оруэлл', genre: 'Роман'},
                {name: 'Собака Баскервилей', author: 'Артур Конан Дойл', genre: 'Детектив'},
                {name: 'Гарри Поттер', author: 'Джоан Роулинг', genre: 'Роман'}
            ]
        }
    },
    methods: {
        lowercase(value) {
            return value.toLowerCase();
        }
    },
    template: '<ul><li v-for="(el, i) in books">{{ i + 1}} - {{ lowercase(el.name) }} - <b>{{ el.author }}</b> / <em>{{ el.genre }}</em></li></ul>'
};

const app = createApp({
    data() {
        return {
            title: "Hello World",
            isVisible: true
        }
    }
});

app.component('main-component', MainComponent);
app.mount("#app");

```

```html
<div id="app">
        <h1 v-if="title != 'Hello World'">{{ title }}</h1>
        <p v-else-if="title == 'Hello World'"><em>{{ title }}</em></p>
        <p v-else><b>{{ title }}</b></p>

        <button @click="isVisible = !isVisible">Скрыть или показать</button>
        <p v-if="isVisible">Некий текст</p>

        <main-component></main-component>
</div>
```
