import math
from math import gcd

x, y, angle = 0, 0, 90
figures = [[(0, 0)]]

def move(step):
    global x, y
    rad = math.radians(angle)
    x += round(step * math.cos(rad))
    y += round(step * math.sin(rad))
    figures[-1].append((x, y))

def turn_right(deg):
    global angle
    angle = (angle - deg) % 360

def turn_left(deg):
    global angle
    angle = (angle + deg) % 360

# ============ НОВЫЕ ФУНКЦИИ ДЛЯ ЛЮБЫХ ФИГУР ============

def polygon_area(pts):
    """Площадь многоугольника по формуле гаусса"""
    n = len(pts)
    area = 0
    for i in range(n):
        j = (i + 1) % n
        area += pts[i][0] * pts[j][1]
        area -= pts[j][0] * pts[i][1]
    return abs(area) / 2

def count_boundary_points(pts):
    """Количество целых точек на границе (рёбрах)"""
    count = 0
    n = len(pts)
    for i in range(n):
        x1, y1 = pts[i]
        x2, y2 = pts[(i + 1) % n]
        count += gcd(abs(x2 - x1), abs(y2 - y1))
    return count

def count_interior_points(pts):
    """Формула Пика: I = A - B/2 + 1"""
    if len(pts) < 3:
        return 0
    area = polygon_area(pts)
    boundary = count_boundary_points(pts)
    return int(area - boundary / 2 + 1)

def count_points_in_rectangle(pts):
    """Для прямоугольников со сторонами параллельными осям"""
    xs = [p[0] for p in pts]
    ys = [p[1] for p in pts]
    return (max(xs) - min(xs) + 1) * (max(ys) - min(ys) + 1)

# ============ ВАШ АЛГОРИТМ (меняется только это) ============

# Пример 1: Одна фигура (ромб)
for _ in range(2):
    turn_right(120)
    move(9)
turn_right(300)
for _ in range(2):
    turn_right(120)

# Пример 2: Несколько фигур (раскомментируйте если нужно)

# ============ ПОДСЧЁТ ТОЧЕК (универсальный) ============

total_points = 0

for fig in figures:
    if len(fig) >= 3:
        # Проверяем, прямоугольник ли это (стороны параллельны осям)
        xs = [p[0] for p in fig]
        ys = [p[1] for p in fig]
        is_rectangle = (len(set(xs)) == 2 or len(set(ys)) == 2)
        
        if is_rectangle:
            # Для прямоугольника можно проще
            points = count_points_in_rectangle(fig)
        else:
            # Для любого многоугольника — формула Пика
            points = count_interior_points(fig)
        
        total_points += points
        print(f"Фигура {figures.index(fig)+1}: {points} точек")

print(f"\nВсего точек: {total_points}")
