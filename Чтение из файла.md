```c++
#include <fstream>

using namespace std;

int main() {
	ifstream fin;
	string path = "myFile.txt";
	fin.open(path);
	if (!fin.is_open()) {
		cout << "Ошибка открытия файла" << endl;
	}
	else {
		cout << "Файл открыт" << endl;
		string str;
		while (!fin.eof()) {
			str = "";
			getline(fin, str);// std
			cout << str << endl;
		}
	}
	fin.close();
	return 0;
}
```
